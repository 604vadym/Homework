git add .                              добавить все изменения в текущей папке и ее подпапках в индекс
git add --all/-A                       добавить вообще все изменения в проекте в индекс (работает из любой папки проекта и сканирует весь локальный репозиторий целиком)
git add -u                             добавит  в индекс изменения только из тех файлов, которые уже отслеживаются (tracked), игнорирует новые (untracked) файлы
git add --ignore-removal .             добавит  в индекс только новые и измененные (modified) файлы (не добавит удаления файлов)
git add -f/--force <file-name>         добавить в индекс даже если файл или папка прописаны в .gitignore
git add *.html                         добавить в индекс все файлы в текущей папке с расширением html
git add dirName/*.html                 добавить в индекс все файлы в папке dirName с расширением html
git add dirName/                       добавить в индекс все файлы в папке dirName
git add "*.html"                       добавить в индекс все файлы в проекте с расширением html
git add -i/--interactive               интерактивный режим индексации (staging) - позволяет точечно манипулировать содержимым индекса (просматривать, добавлять, удалять, смотреть diff и разделять изменения в одном файле на части для создания нескольких коммитов вместо одного (patch))
git add -p                             запустить сразу режим patch интерактивного staging (разделять изменения в одном файле на части для создания нескольких коммитов вместо одного - часть изменений уйдет в индекс, а часть будет только в рабочей директории) - чтобы добавить в индекс не все изменения в файле, а только  нужную часть
git add -p - работа с рабочей директорией и индексом (добавить в индекс только часть изменений в файле)

git commit                             сохранить текущую версию проекта (переместить в репозиторий) - откроется редактор кода (для ввода длинных описаний коммита)
git commit -m "<message>"              коммит с заголовком
git commit -a -m/-am "<message>"       коммит с add и с заголовком, при этом применяется сразу ко всем модифицированным tracked файлам (только к измененным файлам (tracked), но не для новых (untracked))
git commit -m "<message>" <file-name>  коммит с add только конкретного файла или нескольких файлов (минуя стадию добавления в индекс) (только к измененным файлам (tracked), но не для новых (untracked))
git commit --amend                     отредактировать последний коммит (отредактировать описание коммита и добавить содержимое индекса в коммит)
git commit --amend -m "<message>"      отредактировать последний коммит (заменить описание коммита и добавить содержимое индекса в коммит)
git commit --amend --reset-author      отредактировать последний коммит и сделать текущего коммитера автором коммита, обновить авторское время, с возможностю редактировать описание коммита
git commit --amend --no-edit           отредактировать последний коммит (отредактировать изменения в файле коммита, добавив в коммит содержимое индекса, но не описание коммита)
git commit -m "<message>" -m "<message>" сделать коммит с дополнительным описанием
git commit --amend можно делать только последнему коммиту из-за изменения SHA (каждый коммит хранит хеш предыдущего)
git commit --author='name <email>' --date='...' коммит с явным указанием автора и авторской даты
git commit -c <commit hash>           взять описание, информацию об авторе и авторское время для нового коммита из другого коммита с возможностью редактировать описание коммита
git commit -C <commit hash>           взять описание, информацию об авторе и авторское время для нового коммита из другого коммита без возможности редактировать описание коммита
git commit -c/-C <commit hash> --reset-author взять только описание для нового коммита из другого коммита
git commit -v                         коммит с просмотром изменений diff в редакторе (можно прервать коммит если заметил, что нужно что-то изменить, также позволяет точнее делать описание коммита, видя изменения diff)
git commit --fixup <commit hash>      коммит с заплаткой (для последующего rebase -i --autosquash, т.е. этот коммит объединится с указанным даже если укащзанный коммит посередине ветки с удалением заголовка коммита с заплаткой)
git commit --fixup HEAD               коммит с заплаткой к предыдущему коммиту 
git commit --squash <commit hash>     коммит с заплаткой (для последующего rebase -i --autosquash, т.е. этот коммит объединится с указанным даже если укащзанный коммит посередине ветки без удалением заголовка коммита с заплаткой с возможностью редактирования заголовков коммитов в редакторе)

git cat-file -t <commit_hash>         прочитать тип объекта git по хешу объекта
git cat-file -p <commit_hash          прочитать содержимое коммита, дерева и блоба (объекта git) по хешу объекта

git branch <branch name>               создать новую ветку (создается новый указатель на текущий коммит)
git branch <branch-name> <remote (origin обычно)>/<remote branch name> создать локальную ветку на основе удаленной, не перходить в нее и связать их (устанавливает upstream) (трекать, т.е. отслеживать) - после этой команды можно будет делать просто git pull и git push без указания имен веток (можно связать сколько угодно веток)
git branch -u/--set-upstream-to <remote(origin обычно)>/<remote branch name> связать текущую ветку с удаленной (устанавливает upstream) (трекать, т.е. отслеживать) - после этой команды можно будет делать просто git pull и git push без указания имен веток (можно связать сколько угодно веток)
git branch -u/--set-upstream-to <remote(origin обычно)>/<remote branch name> <branch name> связать конкретную существующую локальную ветку с удаленной (устанавливает upstream) (трекать, т.е. отслеживать) - после этой команды можно будет делать просто git pull и git push без указания имен веток (можно связать сколько угодно веток)
git branch --track <branch name> <remote(origin обычно)>/<remote branch name> создать локальную ветку на основе удаленной, не перходить в нее и связать их (устанавливает upstream) (трекать, т.е. отслеживать) - после этой команды можно будет делать просто git pull и git push без указания имен веток (можно связать сколько угодно веток)
git branch --no-track <branch-name> <remote (origin обычно)>/<remote branch name> создать ветку в локальном репозитории на основе ветки на удаленном сервере remote (origin обычно), не перходить в нее и не устанавливать upstream (после этой команды нельзя будет делать просто git pull и git push без указания имен веток)
git branch <branch-name> <commit hash> создать новую ветку и привязать, к ней коммит по его SHA (если нашел утерянный коммит из-за состояния detached и хочу его восстановить или если нужно восстановить удаленную ветку через reflog)
git branch <branch-name> <tag-name>    создать ветку на оснве тега, привязать к ней коммит из тега
git branch -f main <commit-hash>       переместить ветку main на другой коммит (main будет указывать на новый коммит) - используется если по ошибке закоммитили в main и нужно перенести эти коммиты в другую ветку (создаем на этом коммите новую ветку, переключаемся на эту ветку и перемещаем main на предыдущий коммит - в новой ветке будут эти коммиты, а main будет указывать на предыдущий коммит) - лучше, чем reset, т.к. не изменяет файлы в рабочей директории и индексе, а просто переставляет указатель ветки на указанный коммит (но нужно находится в другой ветке)
git branch -f main ORIG_HEAD           отменить последнее действие в ветке (например, merge-коммит или просто только что сделанный коммит в ветке) - лучше, чем reset, т.к. не изменяет файлы в рабочей директории и индексе, а просто переставляет указатель ветки на коммит, на который указывал HEAD перед отменяемым действием (но нужно находится в другой ветке)
git branch --contains <commit hash>    посмотреть в каких ветках находится конкретный коммит (только локальные ветки) (-a или -r если нужны удаленные ветки тоже)
git branch --merged                    посмотреть список веток, которые уже влиты в текущую (их можно смело удалять, так как весь их код уже есть в вашей ветке)
git branch --no-merged                 посмотреть список веток, в которых есть новые коммиты, которых еще нет в текущей ветке
git branch -m/--move <new branch name>  переименовать текущую ветку
git branch -d/--delete <branch name>   удалить определенную ветку локально (текущую ветку удалить нельзя) - удалит ветку только если она уже полностью слита (merged) с текущей или основной веткой
git branch -D/--delete --force <branch name> удалить определенную ветку локально (текущую ветку удалить нельзя) - удалит ветку в любом случае, даже если изменения в ней не были слиты
git bracnh -D --remote <remote(origin обычно)>/<remote branch name> удалить ветку у себя на удаленном сервере remote (origin обычно) (ветка в локальном репозитории удалена не будет)
git branch -vv                         посмотреть как локальные ветки связаны с удалёнными ветками
git branch -v                          показывает последние коммиты в каждой ветке
git branch -d <branch name> следует делать уже после push изменений, чтобы почистить свой локальный репозиторий (чтобы он не замусоривался)

git ls-remote                          показать список всех ссылок (ветки, теги, HEAD) на удаленном репозитории в Github (удаленного серевера, для которого настроен upstream текущей ветки (если он настроен) или удаленного сервера с именем origin (если upstream текущей ветки не настроен), в противном случае ошибка)
git ls-remote <remote(origin обычно)>  показать список всех ссылок (ветки, теги, HEAD) на удаленном репозитории в Github конкретного удаленного сервера

git merge <feature branch name>        слияние другой ветки (feature branch) в текущую ветку (receiving branch) (слияние кода локально в моем репозитории (редактируются блобы)) (перед этим переключится на ветку main)
git merge <remote (origin обычно)>/<branch name> слияние ветки на удаленном сервере remote (origin обычно) с текущей веткой (merge после fetch)

git merge <branch name>                merge с редактированием заголовка merge-коммита в редакторе

git merge --no-edit <branch name>      merge без редактирования заголовка merge-коммита по умолчанию (если оно нас устраивает)

vs
git push --force-with-lease (менее опасна) - лучше всегда использовать ее для самоподстраховки от перезаписи main или чужой ветки
git push --force-with-lease - не использовать в ветке main и чужих ветках (если сделать в чужой или в main, то команда перезатрет историю)
при git push --force-with-lease рекомендуется сделать перед этим git fetch

fetch + rebase + push --force-with-lease для своих веток
git push                               для общих веток

git fetch + git rebase origin/main + git push --force-with-lease - если есть изменения на удаленном сервере

git fetch                              скачать новые объекты из удаленного репозитория, но не применять изменения локально (изменения применяются к ветке remote (origin обычно), но merge с веткой в локальном репозитории не происходит) (если было до этого сделано git remote add)
git fetch                              обновить список всех веток с сервера (если было до этого сделано git remote add)
git fetch - желательно делать как можно чаще (в начале рабочего дня, часто в течение рабочего дня и перед git push)
git fetch <remote (origin обычно)>     fetch из конкретного сервера
git fetch --all                        fetch для всех удаленных репозиториев (если их несколько)
git fetch --all = git remote update
git fetch -p/--prune                   обновит список и сразу удалит из локального списка те ветки, которые на сервере уже стерты

git fetch                              закачивает изменения только в папку .git (использовать после этого git log origin/<branch-name>, git log HEAD..origin/main, git log --all или git diff main origin/main, git diff HEAD..origin/main чтобы просматривать изменения и checkout или reset чтобы применить изменения в нужном коммите или ветке)
vs
git pull                               загружает также в рабочую директорию, а не только в папку .git

git fetch                              загружает все изменения, но можно посмотреть, что было изменено или применить что-то, что тебя интересует (один какой-то коммит или ветку, а не все как в pull)   
vs
git pull                               загружает все изменения в удаленном репозитории и применяет их к локальному репозиторию

git fetch                              выкачивает trees, commits и blobs
vs
git pull                               выкачивает trees, commits и blobs + делает merge ()
git pull = git fetch + git merge

~ - шаги по дереву коммитов по прямой линии
^^ - шаги по дереву коммитов по прямой линии
^2 - шаги по дереву коммитов в сторону (имеет смысл только в merge-коммитах, т.к. в них более одного родителя)
^3 - шаги по дереву коммитов в сторону к третьему родителю (имеет смысл только в octopus-merge-коммитах, т.к. в них более двух родителей)
HEAD не= HEAD~/HEAD не= HEAD^     HEAD^=HEAD^1/HEAD~=HEAD~1     HEAD^^=HEAD~2, HEAD^2 не= HEAD~2
HEAD^2~3 - во второстепенной ветке перейти на 3 коммита назад
HEAD~2^2~3 - на два коммита назад, во второстепенную ветку, в ней еще на три коммита назад
<commit hash>^/<commit hash>~ и т.п. тоже можно
^ и ~ - разница есть только для merge-коммитов - ^ - выбрать конкретного родителя (^1 - в основной ветке (куда вливали через merge), т.е. предыдущий в ветке по отношению к merge-коммиту ^2 - в второстепенной ветке (которую вливали через merge), т.е. последний коммит ветки перед слиянием через merge), ~ - идет строго назад по основной ветке (куда вливали через merge)

git diff                               показать разницу между неподготовленными к коммиту изменениями (в рабочей директории) и индексом (если индекс не пуст) или последним сделанным коммитом (если индекс пуст)

git diff HEAD                          показать вообще все текущие изменения - разницу между неподготовленными к коммиту изменениями (в рабочей директории), а также индексом и последним сделанным коммитом
git diff ORIG_HEAD                     diff к последнему опасному действию
git diff --color-words                 показать разницу не целыми строками, а на уровне отдельных слов (подсвечивает красным какие слова исчезли и зеленым какие слова появились)
git diff --word-diff                   показать разницу не целыми строками, а на уровне отдельных слов (добавляет символьные маркеры изменений и подсвечивает красным какие слова исчезли и зеленым какие слова появились)
git diff <tree-ish>                    показать разницу между неподготовленными к коммиту изменениями (в рабочей директории) и любым tree-ish

~ - шаги по дереву коммитов по прямой линии
^^ - шаги по дереву коммитов по прямой линии
^2 - шаги по дереву коммитов в сторону (имеет смысл только в merge-коммитах, т.к. в них более одного родителя)
^3 - шаги по дереву коммитов в сторону к третьему родителю (имеет смысл только в octopus-merge-коммитах, т.к. в них более двух родителей)
HEAD не= HEAD~/HEAD не= HEAD^     HEAD^=HEAD^1/HEAD~=HEAD~1     HEAD^^=HEAD~2, HEAD^2 не= HEAD~2
HEAD^2~3 - во второстепенной ветке перейти на 3 коммита назад
HEAD~2^2~3 - на два коммита назад, во второстепенную ветку, в ней еще на три коммита назад
<commit hash>^/<commit hash>~ и т.п. тоже можно
^ и ~ - разница есть только для merge-коммитов - ^ - выбрать конкретного родителя (^1 - в основной ветке (куда вливали через merge), т.е. предыдущий в ветке по отношению к merge-коммиту ^2 - в второстепенной ветке (которую вливали через merge), т.е. последний коммит ветки перед слиянием через merge), ~ - идет строго назад по основной ветке (куда вливали через merge)

git stash -p                           stash в режиме patch (можно добавлять в копилку не все изменения в файле, а только выбранные фрагменты, оставлять выбранные фрагменты в рабочей директории и индексе, редактировать изменения и разбивать изменения на части перед отправкой в копилку) - чтобы добавлять в копилку не все изменения в файле, а только нужную часть
git stash -p - работа с рабочей директорией и индексом и копилкой
git checkout -p stash@{n} -- <file-name> / git restore -p --source=stash@{n} <file-name> - checkout/restore в режиме patch (можно доставать из копилки не все изменения в файле, а только выбранные фрагменты, оставлять выбранные фрагменты в копилке, редактировать изменения и разбивать изменения на части перед доставанием из копилки) - чтобы доставать из копилки не все изменения в файле, а только нужную часть
git checkout -p stash@{n} -- <file-name> / git restore -p --source=stash@{n} <file-name> - работа с копилкой и рабочей директорией (и с индексом с флагом --index)
git stash -p --keep-index              stash в режиме patch без удаления из рабочей директории и индекса изменений, которые добавлены в индекс (готовое к коммиту останется в индексе, а остальное уйдет в копилку)

git restore <file-name>                убрать изменения в файле в рабочей директории и заменить на версию из индекса (если делал git add) или на версию из последнего коммита (если не делал git add и индекс пуст)
git restore --source=HEAD <file-name>  убрать изменения в файле в рабочей директории и заменить на версию из последнего коммита
git restore --source=HEAD~3 <file-name> убрать изменения в файле в рабочей директории и заменить на версию три коммита назад
git restore --source <branch name> --staged --worktree <file name> вытащит конкретный файл из другой ветки и положит его в рабочую директорию и в индекс
git restore --staged/--cached <file-name> убрать изменения в файле в индексе (локально файлы не изменяться) и заменить на версию из последнего коммита
git restore --worktree <file-name> убрать изменения в файле рабочей директории и заменить на версию из последнего коммита
git restore --staged --worktree <file-name> убрать изменения в файле и в рабочей директории и в индексе и заменить на версию из последнего коммита
git restore --source <commit hash> --staged --worktree <file-name> убрать изменения в файле в рабочей директории и заменить на тот же определенного коммита (также сразу заносит в индекс)
git restore ., git restore --staged .  то же, что и выше, но для всех файлов в текущей папке и ее подпапках
git restore :/, git restore --staged :/ то же, что и выше, но для всех файлов в репозитории (проекте)
git restore -p                         restore в режиме patch для изменений в рабочей директории (убрать не все изменения в файле, а выборочно фрагментами, оставить выборочно фрагменты, редактировать изменения, разбить изменения на части) - чтобы восстановить в рабочей директории не все изменения в файле, а только нужную часть
git restore -p - работа с рабочей директорией (замена на версию из индекса или указанного коммита)
git restore --staged/--cached -p       restore в режиме patch для изменений только в индексе

git rm <file-name>                     удалить файл из Staging Area и локально (безопасна - удалит только если изменения в рабочей директории и в индексе есть в последнем коммите)
git rm -r <folder-name>                удалить папку из Staging Area и локально
git rm -f/--force <file-name>          удалить файл из Staging Area и локально (небезопасна - удалит даже если изменений нету в последнем коммите)
git rm --staged/--cached <file-name>   удалить файл из Staging Area (удалить из кеша, но не локально)
git rm --staged/--cached <file-name> = git restore --staged/--cached <file-name>

git clean                              удаление неотслеживаемых файлов из директории репозитория (выдаст ошибку, так как сработает запрет на удаление неотслеживаемых файлов из директории репозитория)
git clean -n                           посмотреть, какие неотслеживаемые файлы будут удалены командой clean
git clean -dn                          посмотреть, какие неотслеживаемые файлы  и пустые неотслеживаемые папки будут удалены командой clean
git clean -f/--force                   обход запрета на удаление неотслеживаемых файлов из директории репозитория (удалит только файлы)
git clean -fn                          посмотреть, какие неотслеживаемые файлы будут удалены командой clean с флагом -f/--force 
git clean -fd                          удаляет неотслеживаемые файлы + папки (рекурсивное удаление)
git clean -fdx                         удаляет неотслеживаемые файлы + папки + игнорируемые (.gitignore) (рекурсивное удаление)
git clean -i                           clean в интерактивном режиме (можно удалять выборочно, фильровать по паттерну)

git revert HEAD                        новый коммит, который полностью отменяет изменения последнего сделанного коммита (сразу делает новый реверт коммит - обратный от последнего)
git revert <commit hash>               новый коммит, который полностью отменяет изменения определенного сделанного коммита (сразу делает новый реверт коммит - обратный от указанного)
git revert <commit hash>..<commit hash> revert диапазона коммитов (соседние коммиты и идут в хоронологическом порядке) - не включает первый указанный коммит, создаст набор реверт коммитов
git revert <commit hash>^..<commit hash> revert диапазона коммитов (соседние коммиты и идут в хоронологическом порядке) - включает первый указанный коммит, создаст набор реверт коммитов
git revert <commit hash> <commit hash> <commit hash> revert трех коммитов (порядок коммитов не важен, могут быть не соседними)
git revert <commit hash> <commit hash> <commit hash> - если в коммитах есть зависимости в коде, то нужно указывать коммиты в обратном хронологическом порядке
git revert -n/--no-commit <commit hash> revert без создания автоматического коммита (можно проверить, что все работает и также сделать свой заголовок коммита)
git revert -n/--no-commit <commit hash..<commit hash> revert без создания автоматических коммитов с указанием диапазона коммитов в команде (обратные изменения всех коммитов будут в рабочей директории и в индексе и можно сделать один общий коммит для всех обратных изменений (т.е. слить обратные изменения из нескольких коммитов в один коммит))
git revert -n/--no-commit <commit hash> <commit hash> revert без создания автоматических коммитов с указанием сразу нескольких коммитов в команде (обратные изменения всех коммитов будут в рабочей директории и в индексе и можно сделать один общий коммит для всех обратных изменений (т.е. слить обратные изменения из нескольких коммитов в один коммит))
git revert --no-edit <commit hash>..<commit hash> revert с автоматическим созданием стандартного revert заголовка комммита (полезно для диапазона или набора коммитов (т.к. по умолчанию нужно вводить в редакторе заголовок для каждого коммита), а также если просто устраивает стандартный revert заголовок коммита даже для одного коммита)
git revert --continue                  продолжить после разрешения конфликтов вручную
git revert --abort                     отмена всего revert (при возникновении конфликта) - отмена всех коммитов отмены
git revert--quit                       остановить revert, но без отмены всех коммитов отмены - их оставить, но дальше revert не делать
git revert --skip                      пропустить коммит и не применять к нему revert (используется при диапазоне коммитов при конфликтах)
git revert -m 1 <commit hash>          отмена merge-коммита (даже если он уже был отправлен в общий репозиторий через push), -m 1 - считать первую ветку (т.е. в которую вливали) основной и оставить состояние той ветки, в которую вливали (будет новый коммит, а не переписывыание истории, используется, чтобы не испортить историю другим разработчикам, которые успели загрузить себе наш merge-коммит) - обычный revert на merge-коммите выдаст ошибку
git revert -m 2 <commit hash> - (опасно) не перепутать!!! использовать git status!!! -m 2 применяется только если вливали, например main в feature - откат к состоянию вливаемой ветки (если ошибиться и сделать -m 2 при слиянии feature в main, то ошибки не будет, но весь код в main будет удален и заменен на состояние ветки feature)
git revert -x <commit hash>            указать в сообщении коммита хеш того коммита, который был отменен при revert (полезно для отслеживания оригинального коммита, который был отменен через revert)

git revert - следует делать вместо git reset если уже сделал push коммита (чтобы не испортить историю другим разработчикам)
git revert -m 1 <commit hash> - чтобы потом при слиянии веток не потерять изменения, отмененные при реверте merge-коммита (git не знает, что были отменены изменения при слиянии, так как merge-коммит не удален, а отменен через revert) можно сделать один из следующих вариантов и уже после этого делать merge ветки в текущую ветку:
- сделать rebase --onto ветки - не нужен реверт на реверт и история будет легкочитаемая и линейная
  проще и предпочтительнее, но работает только с нашими влитыми ветками, т.к. rebase чужих веток нужно избегать из-за переписывания истории
     - еще есть вариант - git rebase <base commit hash> --no-ff (это делает копию текущей ветки (вливаемой)) + git checkout <target branch name> + git merge <branch name>
- сделать revert коммита отмены merge-коммита (т.е. реверт на реверт - хуже чем rebase --onto, но лучше и надежнее, чем cherry pick) или второй вариант (хуже всего) cherry-pick -m 1 отмененного merge-коммита
  сложнее, запутаннее история и менее предпочтительнее, но работает с чужими влитыми ветками (если был revert merge-коммита слияния чужой ветки)
git revert -m 1 <commit hash> - лучше избегать, так как потом это может усложнить чтение истории и поиск источника изменений - merge-коммит -> реверт merge-коммита -> реверт на реверт merge-коммита -> новый merge-коммит + могут быть коммиты между каждым из этих действий
git revert -m 1 <commit hash> - альтернатива (если изменений немного) - это просто вручную исправить код или вручную удалить лишнее после слияния (т.е. то, что сделал бы revert сделать вручную) и сделать обычный новый коммит

git filter-branch --tree-filter 'shell command' HEAD применить команду shell ко всем коммитам в ветке на которую указывает HEAD
git filter-branch --tree-filter 'rm -f <file-name>' HEAD удалить конкретный файл из каждого коммита в текущей ветке (будут созданы новые коммиты)
git filter-branch --subdirectory-filter <folder-name> HEAD сделать содержимое конкретной паки новым корнем отдельного самостоятельного репозитория (с сохранением истории правок только для этой папки и ее содержимого)
git filter-branch - переписывает историю (можно делать для своих локальных веток, но для общих и запушенных своих лучше не делать или быть аккуратным,)

git switch <branch name>               переключить ветку (checkout только для веток) (перемещение HEAD)
git switch -f/--force <branch name>    переключить ветку c удалением изменений в рабочей диреткории и индексе и заменой на то, что в другой ветке (без возможности восстановить)
git switch -m/--merge <branch name>    переключить ветку со слиянием (автоматическим или вручную) изменений в рабочей диреткории и индексе с версиями в другой ветке
git switch -c/--create <branch name>   создать ветку и переключиться на нее (незакоммиченные изменения переходят в новую ветку)
git switch -c <branch name> <tag-name>/<commit hash> создать ветку на основе тега или коммита
git switch -c -m <branch name> <tag-name>/<commit hash> создать ветку на основе тега или коммита со слиянием merge текущих незакомиченных изменений
git switch -c <branch name> HEAD~3     создать новую ветку на основе состояние текущей ветки три коммита назад
git switch -c -m <branch name> HEAD~3  git checkout -b -m <branch name> HEAD~3 создать новую ветку на основе состояние текущей ветки три коммита назад со слиянием
git switch -c <branch name/name>       создать ветку с вложенным именем (использовать папку в названии)
git switch -c <branch name> --track <remote(origin обычно)>/<remote branch name> создать локальную ветку на основе удаленной, перейти в нее и связать их (устанавливает upstream) (трекать, т.е. отслеживать) - после этой команды можно будет делать просто git pull и git push без указания имен веток (можно связать сколько угодно веток)
git switch -c <branch name> --track <remote(origin обычно)>/<remote branch name> = git switch -c <branch name> <remote(origin обычно)>/<remote branch name>
git switch -c <branch name> --no-track <remote(origin обычно)>/<remote branch name> создать локальную ветку на основе удаленной, перейти в нее и не связывать их (не устанавливает upstream) (не трекать, т.е. не отслеживать)
git switch --detach <commit hash>/<tag-name> переключиться на конкретный коммит или тег с помощью switch даже если HEAD будет detached

git checkout                           переместится между версиями проекта (переместится HEAD)
git checkout -- <file-name> (не использовать - git restore лучше) убрать изменения в файле в рабочей директории и заменить на индекс (если индекс не пуст) или на последний коммит (если индекс пуст) (отмена локальных изменений в конкретном файле)
git checkout master/main               вернуться в ветку master/main
git checkout <branch name>             перейти в другую локальную ветку (если есть конфликтующие с другой веткой незакоммиченные изменения в рабочей директории или в индексе, то выдаст ошибку, а если есть неконфликтующие изменения, то git протащит(перенесет) их при переключении и оставит в рабочей директории или в индексе)
git checkout <branch name>             если локальной ветки такой нету, а удаленная есть, то создаст локальную ветку с таким же названием, которая будет указывать на удаленную ветку (устанавливает upstream) (трекать, т.е. отслеживать) и дать ей имя такое же как у удаленной ветки и перейти в нее - после этой команды можно будет делать просто git pull и git push без указания имен веток
git checkout <branch name> (не использовать - git switch лучше) переход в определенную ветку
git checkout <branch name> -- <file-name> вытащит конкретный файл из другой ветки и положит его в рабочую директорию и в индекс
git checkout -f HEAD / git checkout -f принудительно удалить все незакоммиченные изменения в рабочей директории или в индексе без возможности восстановления (даже reflog не поможет) - перепишет их версиями из того объета, на который указывает HEAD (обычно последний коммит текущей ветки)
git checkout -- <file-name>            перенести изменения в конкретном файле из индекса в рабочую директорию
git checkout HEAD  -- <file-name>      перенести изменения в конкретном файле из состояния проекта, на который указывает HEAD (обычно последний коммит в текущей ветке) в индекс и рабочую директорию
git checkout -f <branch name>          принудительно перейти в другую ветку и удалить все незакоммиченные изменения в рабочей директории или в индексе без возможности восстановления (даже reflog не поможет)
git checkout -m <branch name>          переключить ветку со слиянием (автоматическим или вручную) изменений в рабочей диреткории и индексе с версиями в другой ветке
git checkout -b <branch name>          создать новую ветку и перейти в нее (незакоммиченные изменения переходят в новую ветку)
git checkout -b <new branch name> <remote (origin обычно)>/<remote branch name> создать локальную ветку, которая будет указывать на удаленную ветку (устанавливает upstream) (трекать, т.е. отслеживать) и дать ей новое указанное имя и перейти в нее - после этой команды можно будет делать просто git pull и git push без указания имен веток
git checkout -b <branch name> <tag-name>/<commit hash> создать новую ветку и привязать к ней тег или коммит по его SHA и перейти в нее (если нашел утерянный коммит из-за состояния detached и хочу его восстановить или если нужно восстановить удаленную ветку через reflog)
git checkout -b -m <branch name> <tag-name>/<commit hash> создать ветку на основе тега или коммита со слиянием merge текущих незакомиченных изменений
git checkout -B -m <branch name> <commit hash> если ветка с таким именем уже есть, то она будет передвинута на указанный коммит (будет указывать на него) и переклюимся на эту ветку если ветки с таким именем нет, то будет создана ветка на основе коуретного коммита - используется если по ошибке закоммитили в ветку и нужно перенести эти коммиты в другую ветку (создаем на этом коммите новую ветку, переключаемся на новую ветку ветку и перемещаем ветку на предыдущий коммит - в новой ветке будут эти коммиты, а ветка будет указывать на предыдущий коммит)
git checkout -b <branch name> HEAD~3   создать новую ветку на основе состояние текущей ветки три коммита назад
git checkout -b -m <branch name> HEAD~3 создать новую ветку на основе состояние текущей ветки три коммита назад со слиянием
git checkout --track <remote (origin обычно)>/<remote branch name> создать локальную ветку, которая будет указывать на удаленную ветку (устанавливает upstream) (трекать, т.е. отслеживать) и дать ей имя такое же как у удаленной ветки и перейти в нее - после этой команды можно будет делать просто git pull и git push без указания имен веток
git checkout -b <branch name> --track <remote(origin обычно)>/<remote branch name> создать локальную ветку на основе удаленной, перейти в нее и связать их (устанавливает upstream) (трекать, т.е. отслеживать) - после этой команды можно будет делать просто git pull и git push без указания имен веток (можно связать сколько угодно веток)

git checkout                           перейти к определенной версии (перемещение из репозитория в рабочую директорию) перемещает HEAD, но не ветку делает состояние detached HEAD

git reset --mixed HEAD~2 / git reset --mixed HEAD^^ вернуться на два коммита назад (переместиться HEAD и ветка) (без удаления изменений в рабочей директории)  - перемещает два последних коммита в рабочую директорию, также изменения в рабочей директории остаются, индекс становится как в коммите, на который перешли

git reset --hard <remote (origin обычно)>/<branch name> удалить все незакомиченные изменения и сделать текущую ветку точной копией удаленной ветки (коммиты в текущей ветке, которых нет в целевой ветке исчезнут)
git reset -p                           reset в режиме patch (работает только с флагом --mixed) для изменений в индексе (убрать не все изменения в файле, а выборочно фрагментами, оставить выборочно фрагменты, редактировать изменения, разбить изменения на части) - чтобы разделить изменения в файле в индексе на несколько коммитов, убрать лишние для коммита изменения в файле из индекса
git reset -p - работа с индексом (то, что уже в индексе) - убрать из индекса, отредактировать то, что в индексе
git reset --keep                       сохраняет изменения в индексе, если они не конфликтуют с целевым коммитом, обновляет файлы в локальной директории под целевой коммит (в данном случае последний) если при этом не возникает конфликтов при объединении локальных изменений и состояния файлов в целевом коммите - это тот же --hard, но с сохранением локальных изменений, но при этом без сохранения состояния промежуточных коммитов (в отличие от --soft и --mixed)
--soft                                 код остается в индексе и на диске (готов к коммиту) (также все коммиты после указанного переводятся в зону staged (индекс), но не удаляются, содержимое индекса объединяется с коммитами после указанного)
--mixed                                код остается на диске, но вылетает из индекса и переносится на диск (нужно снова делать git add) (по умолчанию) (также все коммиты после указанного переводятся в рабочую директорию, но не удаляются, изменения в рабочей диретории объединяются с коммитами после указанного, а индекс становятся такими как в коммите, на который перешли)
--hard                                 код удаляется отовсюду - и с диска и с индекса (возвращает проект к указанному коммиту - рабочая директория и индекс становятся такими как в коммите, на который перешли, при этом полностью удаляет все коммиты после указанного)
--keep                                 код остается на диске, а индекс переносится на диск (но все коммиты после указанного не сохраняются, а содержимое рабочей директории и индекса объединяется с целевым коммитом если нет конфликтов, если конфликты есть, то будет ошибка, индекс становятся такими как в коммите, на который перешли)
--soft                                 можно использовать для объединения (squash) последних коммитов в один или с текущими изменениями в индексе
--mixed                                можно использовать для редактирования последних коммитов и также для объединения (squash) последних коммитов в один или с текущими изменениями в индексе
--hard                                 можно использовать для удаления последних коммитов и/или очищения рабочей директории и индекса
--keep                                 можно использовать для переноса локальных изменений в другую ветку (если делал изменения по ошибке не в той ветке) или к состоянию предыдущих коммитов в той же ветке (если сделал по ошибке промежуточные коммиты не в той ветке, через reflog (в новую ветку) или cherry-pick (в другую существующую) можно перенести промежуточные коммиты в другую ветку)
--keep                                 нужна, чтобы переключить состояние проекта, сохранив свои наработки (вместо stash сразу переносит измененияв целевое состояние проетка если нет конфликтов при переносе) - это тот же --hard, но с сохранением локальных изменений, не сохраняя состояния промежуточных коммитов
--hard vs --keep                       --hard не переносит незакоммиченные изменения, а --keep переносит незакомммиченные изменения

~ - шаги по дереву коммитов по прямой линии
^^ - шаги по дереву коммитов по прямой линии
^2 - шаги по дереву коммитов в сторону (имеет смысл только в merge-коммитах, т.к. в них более одного родителя)
^3 - шаги по дереву коммитов в сторону к третьему родителю (имеет смысл только в octopus-merge-коммитах, т.к. в них более двух родителей)
HEAD не= HEAD~/HEAD не= HEAD^     HEAD^=HEAD^1/HEAD~=HEAD~1     HEAD^^=HEAD~2, HEAD^2 не= HEAD~2
HEAD^2~3 - во второстепенной ветке перейти на 3 коммита назад
HEAD~2^2~3 - на два коммита назад, во второстепенную ветку, в ней еще на три коммита назад
<commit hash>^/<commit hash>~ и т.п. тоже можно
^ и ~ - разница есть только для merge-коммитов - ^ - выбрать конкретного родителя (^1 - в основной ветке (куда вливали через merge), т.е. предыдущий в ветке по отношению к merge-коммиту ^2 - в второстепенной ветке (которую вливали через merge), т.е. последний коммит ветки перед слиянием через merge), ~ - идет строго назад по основной ветке (куда вливали через merge)

git rebase <branch name (main/master обычно)> перекладывание новых/уникальных коммитов из одной ветки поверх верхушки другой ветки (перебазируются коммиты из текущей ветки поверх коммитов в указанной в команде ветки, указанная в команде ветка при этом не изменится вообще (ход коммитов до git merge останется тот же), перебазируется только текущая ветка (в нее копируются недостающие коммиты из другой ветки и поврех них перекладываются новые/уникальные коммиты этой ветки))
git rebase <branch name> <branch name> rebase уникальных коммитов из второй ветки поверх коммитов в первой ветке (переключись на указанную вторую ветку и сделай rebase всех коммитов в этой ветке) - HEAD будет на перебазированной второй ветке
git rebase <branch name> <commit hash> rebase уникальных коммитов, которые идут начиная с указанного коммита (включая его) из другой ветки поверх коммитов в первой ветке (переключись на указанный коммит и сделай rebase всех коммитов после указанного коммита) - HEAD будет на перебазированной второй ветке в состоянии detached HEAD (придется делать еще git branch -f main HEAD)
git rebase origin/main                 выполнять этот rebase после fetch
git am --show-current-patch            посмотреть код из коммита, где произошел конфликт при rebase
git rebase --rebase-merges             rebase с сохранением структуры слияний внутри переносимой ветки (merge-коммиты, которые уже были в ветке, к которой применяется интерактивный rebase, не будут выпрямлены) - выполняется автоматически
git rebase --continue                  использовать для завершения процесса перекладывания (перебазирования) после ручных правок при конфликтах и git add
git rebase --abort                     остановить процесс перебазирования и вернуть ветку в исходное состояние (до того, как ввели команду rebase) (полностью отменить всё и вернуться в состояние до начала перебазирования)
git rebase --skip (опасная - лучше не использовать) пропустить коммит, который вызывает конфликт (полностью теряются изменения, которые были в этом коммите)
git rebase --quit                      прекратить rebase, но не удалять уже перебазированные коммиты и оставить рабочую область как есть (ветка останется в подвешенном состоянии, HEAD не вернется назад)
git rebase <base commit hash> --no-ff  сделать копию текущей ветки начиная с указанного коммита (используется если были revert merge-коммитов и нужно не потерять отмененные коммиты и влить их заново в другую ветку)
git rebase -x <command> <branch name>  запуск команды после каждого перебазированного коммита - обычно для выполнения тестов при перебазировании, чтобы не было неработающего кода после перебазирования из-за семантического конфликта (если команда вернет 0 - все ОК, если не 0, то rebase останавливается с возможностью исправления конфликта в коммите, вызвавшем невыполнение команды/тестов - при этом этот коммит нужно будет исправить через commit --amend --no-edit и дальше rebase --continue)
git rebase --rebase-merges <branch name> rebase с копированием merge-коммита (без пропуска merge-коммита как это по умолчанию происходит при rebase) - использовать если в merge коммите был добавлен код при разрешении конфликта слияния или был merge --no-commit с добавлением кода, посзволяет не потерять этот добавленный код (при это истори будет нелинейной, так структура ветвления сохранится вместе с merge-коммитом, а коммиты из вливаемое ветки не будут перебазированы, вместо этого они будут сохранены как коммиты-родители в перебазированном merge-коммите) (обычный rebase merge-коммиты пропускает)

git rebase - для избегания merge-коммитов и чтобы была линейная история в репозитории или если для работы собственных изменений нужны чужие изменения (чужой коммит) (в собственных изменениях используются чужие изменения)
git rebase - после него нужно сделать merge веток (просто он будет всегда fast-forward) и после этого уже можно удалить ненужную ветку (после merge и rebase удаляется только имя ветки, структура ветвелния (--graph) при merge остается, а при rebase структура становится линейной)
git rebase - никогда не делать для коммитов на ветвях, которые уже были отправлены в общий удаленный репозиторий через push (т.к. это перепишет историю другим разработчикам - изменится SHA коммита)
git rebase - хорош для своих локальных веток, но для общих веток (которые есть в remote (origin обычно)) плох, т.к. может переписать историю
git rebase - обычно делать нужно находясь в своей ветке и указывать аргументом целвевую ветку (чаще всего main)
git rebase - делать, находясь в своей ветке, а не в main - делать git branch + swith если нужно
git rebase - при rebase merge-коммиты просто пропускаются, дже если в них был добавлен код при разрешении конфликта слияния или был merge --no-commit с добавлением кода это решается флагом --rebase-merges
git rebase - при rebase если происходят конфликты в перебазированных коммитах (или даже был merge-конфликт в merge-коммите (в таком случае он не пропускается, а дается повторно разрешить)), то нужно повторно разрешить этот конфликт и так повторно для каждого следующего перебазирования коммита - решается с помощью rerere
git rebase - перебазирует только новые/уникальные коммиты, т.е. которые есть в одной ветке, но нет в другой, без дублирования (просто пропускает коммиты-дубли)

git rebase                             если в ветке мало изменений
vs
git merge                              если в ветке много изменений (или если в проекте не просят использовать rebase для линейной истории)

git rebase --onto <target branch name> <commit hash> перебазирование части ветки начиная с конкретного коммита в текущей ветке в указанную target ветку (не включает конкретный указанный коммит)
git rebase --onto <target branch name> <branch name> перебазирование части ветки начиная с конкретного коммита в текущей ветке (в данном случае момент ответвления) в указанную target ветку (не включает последний коммит в указанной ветке - перебазирует все коммиты, которые идут в текущей ветке после указанной ветки) - если по ошибке ответвились не от той ветки
git rebase --onto <target branch name> <branch name1> <branch name2> перебазирование ветки <branch name2>, которая ответвилась от <branch name1> поверх коммитов в <target branch name> (перебазирует все коммиты в <branch name1>, начинающиеся в ответвившейся от нее ветке <branch name2> поверх коммитов в <target branch name>)
git rebase --onto <target branch name> <branch name1> <branch name2> не перебазирует все, что было до <branch name1>, перебазирует все, что есть в <branch name2>, но чего нет в <branch name1>
git rebase --onto <target branch name> <branch name1> <branch name2> = checkout <branch name2> + rebase --onto <target branch name> <branch name1> - если по ошибке ответвились не от той ветки

git rebase --onto <target branch name> <base commit hash> <branch name>/<finish commit hash> перебазирование ветки если были реверты merge-коммитов (обычные rebase в таком случае перенесет коммиты ветки неправильно - он перенесет только коммиты после последнего расхождения (тот самый отмененный merge-коммит), а нужны в таком случае все коммиты в ветке) - использовать вместо реверта на реверт merge-коммита
<base commit hash> - от какого коммита начать перебазирование (обычно базовый коммит, где произошло начальное разветвление)
<branch name>/<finish commit hash> - каким коммитом закончить перебазирование (может быть и пустым и HEAD и т.п. если находимся в ветке, которую перебазируем)
git rebase --onto <target branch name> <base commit hash> <branch name>/<finish commit hash> - в случае если был revert merge-коммита, то эта команда перенесет наоборот не часть ветки, а как раз всю ветку (тогда как обычный rebase в таком случае наоборот перенесет только часть ветки после момента того отмененного merge-коммита, так как переносит только уникальные коммиты и посчитает, что раз был merge-коммит, то коммиты перед ним не уникальны, а уже слиты (и он не понимает, что дальше был revert этого merge-коммита))

git rebase -i/--interactive            интерактивный rebase (можно перебазировать коммиты как относительно той же ветки так и относительно другой ветки) (без аргумента не сработает)
git rebase -i HEAD~3                   интерактивный rebase трех последних коммитов (три коммита переместятся во временную зону и после этого с ними произойдет то, что мы укажем вручную командами в инструкции для интерактивного rebase) (перебазирует коммиты в текущей ветке относительно коммита в той же ветке)
git rebase -i <commit hash>            интерактивный rebase всех коммитов, идущих после указанного
git rebase -i <commit hash>^           интерактивный rebase всех коммитов, идущих после указанного, включая указанный
git rebase -i <branch name>            интерактивный rebase только тех коммитов в текущей ветке, которых нет в указанной (перебазирует относительно верхушки (последнего коммита) в указанной ветке)
git rebase -i --root                   интерактивный rebase абсолютно всех коммитов в текущей ветке, включая самый первый (корневой)
git rebase -i --autosquash             интерактивное перебазирование с слиянием коммитов с заплаткой с целевыми коммитами
git rebase -i --rebase-merges          интерактивный rebase с сохранением или пересборкой (изменением) структуры слияний внутри переносимой ветки (merge-коммиты, которые уже были в ветке, к которой применяется интерактивный rebase, не будут выпрямлены) - выполняется вручную
git rebase -i --rebase-merges          можно пересобирать (изменять) структуру слияний и ветвлений не только для merge-коммитов, а для любых ветвлений, можно делать, что угодной в плане ветвлений и пересборки (изменения) графа (дерева) коммитов
git rebase -i --rebase-merges - конструктор и редактор графа (дерева) коммитов (в отличие от простого git rebase --rebase-merges, с помощью которого только избегают выпрямления merge-коммитов)
git rebase --continue                  использовать для завершения процесса интерактивного перебазирования (перекладывания) после ручных правок при конфликтах и git add
git rebase --abort                     отменить интерактивный rebase
git rebase --skip (опасная - лучше не использовать) пропустить коммит, который вызывает конфликт (полностью теряются изменения, которые были в этом коммите)
git rebase --quit                      прекратить интерактивный rebase, но оставить рабочую область как есть (ветка останется в подвешенном состоянии, HEAD не вернется назад)
git rebase --edit-todo                 проверить или изменить действия в процессе интерактивного перебазирования
pick (p) - оставить как есть (просто применить коммит к ветке как он есть)
reword (r) — изменить сообщение коммита
edit (e) — изменить содержимого коммита (через commit --amend) или разделить на несколько коммитов (через несколько add и несколько commit)
squash (s) — склеить этот коммит с предыдущим (склеить содержимое, т.е. код) (вручную в новом окне можно будет ввести любое сообщение коммита или использовать старые, редактировать их)) (объединить два коммита в один)
fixup (f) — склеить этот коммит с предыдущим (склеить содержимое, т.е. код), но удалить сообщение этого коммита (останется сообщение предыдущего коммита)
exec (x) - запустить команду после создания коммита (например, тест)
drop (d) — удалить коммит
также можно поменять коммиты местами (в инструкции для интерактивного rebase) и разделить коммит на несколько коммитов (использовать edit (e))

git rebase -i с HEAD работает с коммитами, которые идут хронологически после того коммита, который указан
git rebase -i HEAD - не сработает вообще, git rebase -i HEAD^ - применится к последнему коммиту (к HEAD, а не к HEAD^)

git cherry-pick <branch name>          копирует последний коммит из конкретной ветки в текущую (состояние кода последнего в ветке коммита) (копирует последний коммит в указанной ветке поверх последнего коммита в текущей ветке, путем создания нового коммита в текущей ветке с тем же сообщением и изменениями, что в коммите указанной ветки), при этом в текущей ветке создается новый коммит, но копирующий целевой
git cherry-pick <commit hash>          копирует конкретный коммит из другой ветки в текущую (состояние кода указанного конкретного коммита) или даже коммит, сделанный в состоянии detached HEAD (на него не указывает ни одна ветка, т.е. он ни в одной ветке) (копирует конкретный коммит поверх последнего коммита в текущей ветке, путем создания нового коммита в текущей ветке с тем же сообщением и изменениями, что в указанном коммите)
git cherry-pick <commit hash> <commit hash> <commit hash> копирует три конкретных коммита из другой ветки в текущую
git cherry-pick <commit hash> <commit hash> <commit hash> - если в коммитах есть зависимости в коде, то нужно указывать коммиты в хронологическом порядке
git cherry-pick <commit hash>..<commit hash> cherry-pick диапазона (набора/группы) коммитов (первый коммит не включается в выборку)
git cherry-pick <commit hash>^..<commit hash> cherry-pick диапазона коммитов (набора/группы) (первый коммит включается в выборку)
git cherry-pick <branch name>..<branch name> cherry-pick всех коммитов из второй ветки, которых нету в первой ветке
git cherry-pick -n/--no-commit <commit hash> cherry-pick без создания автоматического коммита (можно что-то отредактировать только в одном коммите (т.к. изменения будут в индексе) и сделать новый коммит)
git cherry-pick -n/--no-commit <commit hash> <commit hash> cherry-pick без создания автоматических коммитов с указанием сразу нескольких коммитов в команде (можно скопировать в рабочую директорию и в индекс несколько коммитов подряд, что-то отредактировать и только потом сделать один общий коммит для всех изменений (т.е. слить изменения из нескольких коммитов в один коммит, если нужно добавив какие-то изменения в код))
git cherry-pick -n/--no-commit <commit hash>..<commit hash> cherry-pick без создания автоматических коммитов с указанием диапазона коммитов в команде (можно скопировать в рабочую директорию и в индекс диапазон коммитов, что-то отредактировать и только потом сделать один общий коммит для всех изменений (т.е. слить изменения из диапазона коммитов в один коммит, если нужно добавив какие-то изменения в код))
git cherry-pick -e/--edit <commit hash> cherry-pick с изменением сообщения коммита
git cherry-pick --continue             продолжить после разрешения конфликтов вручную
git cherry-pick --abort                отмена всего cherry-pick (при возникновении конфликта) - отмена всех скопированных коммитов
git cherry-pick --quit                 остановить cherry-pick, но без отмены всех скопированных коммитов - их оставить, но дальше cherry-pick не делать
git cherry-pick --skip                 пропустить коммит и не применять к нему cherry-pick (используется при диапазоне коммитов при конфликтах)
git cherry-pick -x <commit hash>       указать в сообщении коммита хеш того коммита, который был скопирован при cherry-pick (полезно для отслеживания того, откуда пришел входной комит)
git cherry-pick -s/--signoff <commit hash> указать в сообщении коммита имя того пользователя, кто совершил cherry-pick (имя автора (author) копируется при cherry-pick, поэтому может потребоваться эта команда)
cherry-pick -m 1 <commit hash>         перенос результата слияния в другую ветку или cherry-pick отмененного merge-коммита (но реверт на реверт лучше и надежнее)

git blame <file-name>                  посмотреть всех авторов кода для всего файла построчно с хешами коммитов и датами
git blame -L 2,5 <file-name>           найти автора кода и в каком коммите - код со второй по пятую строку в конкретном файле
git blame -L 100,+15 <file-name>       найти автора кода и в каком коммите - код с сотой строки и еще пятнадцать строк вниз в конкретном файле
git blame -L :funtion-name: <file-name> найти автора кода и в каком коммите - конкретную функцию
git blame -С -L 2,5 <file-name>        найти автора кода и в каком коммите - ищет копии в других файлах, которые были изменены в том же самом коммите, что и текущий (если на самом деле код был просто скопирован или перенесен из другого файла и у него на самом деле другой оригинальный автор)
git blame -С -С -L 2,5 <file-name>     найти автора кода и в каком коммите - ищет копии в коммите, который создал файл и в коммите-родителе коммита, в котором файл был создан во всех файлах, которые были в проекте (если на самом деле код был написан автором файла или перенесен в него из другого файла при создании файла и у него на самом деле другой оригинальный автор)
git blame -С -С -С -L 2,5 <file-name>  найти автора кода и в каком коммите - ищет копии вообще по всем файлам во всей истории репозитория (самый глубокий поиск, найдет автора когда бы и где бы ни было это изменение, но работает медленнее всего) (если на самом деле код был скопирован из другого файла и у него на самом деле другой оригинальный автор)

ветка - по простому указатель на коммит (внутри устроен как файл с хешем коммита) (можно перемещать с помощью switch и reset)
HEAD - по простому указатель на текущую ветку (но также в стостоянии detached может указывать и на коммит), можно перемещать с помощью checkout, reset и switch (внутри устроен как файл с хешем ветки или даже коммита)
ORIG_HEAD - по простому указатель на состояние проекта до последнего опасного действия (merge, rebase или reset) - для undo
MERGE_HEAD - по простому указатель на верхушку сливаемой ветки (появляется только при merge конфликте)
origin/HEAD - по простому специальный указатель, который сообщает, какая ветка в удаленном репозитории (обычно origin) считается веткой по умолчанию (основной)

git check-ignore -v <path name>         проверить попадает ли файл или папка под какой-то из шаблонов в .gitignore и если да, то под какой
https://github.com/github/gitignore - коллекция файлов .gitignore с шаблонами под различные языки программирования и технологии

.gitkeep - это пустой файл-пустышка, который создают вручную, чтобы Git видел и сохранял в репозитории пустые папки если это нужно (без этого Git не видит пустые директории)

.gitattributes - это конфигурационный файл для настройки того, как git сравнивает файлы (diff), объединяет файлы (merge) как обрабатывает окончания строк(text)
.gitattributes - хранить его лучше как и .gitignore в корне проекта и на удаленном сервере, чтобы он был у всех и при этом одинаковый
git help attributes                    посмотреть справку по атрибутам 

*.html diff=html - в .gitattributes можно добавить эту сртроку (драйвер diff для .html файлов) для более корректного сравнения .html файлов через diff (пждя правильного восприятия слов указанного языка и заголовков hunk)
*.css diff=css - в .gitattributes можно добавить эту сртроку (драйвер diff для .css файлов) для более корректного сравнения .css файлов через diff
*.png -diff/*.svg -diff - сделать так, чтобы diff драйвер воспринимал указанные форматы файлов как бинарные, а не текстовые
*.exe binary - атрибут-псевдоним трех настроек -text -merge -diff

.gitattributes - пример кроссплатформенной настройки (у всех как Windows так и Linux) нормализации переводов строк и распознавания текстовых и бинарных файлов
* text=auto

# force text
*.html text
*.js text
*.css text

# Windows files (need CRLF)
*.sln eol=crlf
*.bat eol=crlf

# Unix files (need LF)
*.sh eol=lf

# binary files
*.jpg -text
*.gif -text
# ...

.gitattributes - пример некроссплатформенной настройки (у всех Linux)
* text=auto eol=lf
# ...

.gitattributes - пример некроссплатформенной настройки (у всех Windows)
* -text
# ...
